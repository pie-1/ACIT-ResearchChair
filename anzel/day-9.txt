Topic learned: C++ (Function Overloading and Overriding)

Summary:
        Function overriding is a concept of object-oriented programming which allows a derived class to redefine a function that is already defined in the base class.
        Function overloading is a concept of OOP which allows the same named function to be called using different methods.

//Q7

#include<iostream>                  //function Overriding
using namespace std;

class Employee
{

    public : virtual void salary()
    {
        cout<<"Base class Salary:"<<endl;
    }

};

class FullTimeEmployee : public Employee
{
    public: void salary()
    {
        cout<<"Your salary is : Rs50,000"<<endl;
    } 
};

class PartTimeEmployee: public Employee
{
     public: void salary(){
        cout<<"Your salary is : Rs30,000"<<endl;
    } 
};


int main()
{
    Employee *e;
    FullTimeEmployee f;
    e=&f;
    (*e).salary();

    PartTimeEmployee p;
    e=&p;
    (*e).salary();

    return 0;
}
